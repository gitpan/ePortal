'/' . $st->Clicks; } $h{$st->date} = $st->daily_hits . '/' . $st->daily_visitor; } push @data, { %h } if exists $h{title}; # Сбор статистики по месяцам $st = new ePortal::ThePersistent::Support( SQL => q{SELECT c.Title, c.Clicks, c.Hits, date_format(s.date, '%m.%Y') as month, sum(s.hits) as monthly_hits, count(distinct s.visitor) as monthly_visitor from Statistics s left join Catalog c on s.catalog_id=c.id}, OrderBy => 'c.title, month', GroupBy => 'catalog_id, month', ); $st->restore_where(); while($st->restore_next) { foreach my $d (@data) { if ( $d->{title} eq $st->Title ) { $d->{$st->month} = $st->monthly_hits . '/' . $st->monthly_visitor; last; } } }
